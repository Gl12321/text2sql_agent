import chromadb
from typing import Any, Optional
from src.core.logger import setup_logger
from pathlib import Path

logger = setup_logger("retriever")

class SchemaRetriever:
    def __init__(self, path: str = "./data/chroma_db"):
        """
        Инициализация локального векторного хранилища.
        """
        # Гарантируем наличие папки для БД
        Path(path).mkdir(parents=True, exist_ok=True)

        self.client = chromadb.PersistentClient(path=path)

        # Коллекция для DDL (строки для LLM)
        self.table_collection = self.client.get_or_create_collection(
            name="tables_detailed",
            metadata={"hnsw:space": "cosine"}
        )

        # Коллекция для саммари схем (на будущее)
        self.schema_collection = self.client.get_or_create_collection(
            name="schemas_summary",
            metadata={"hnsw:space": "cosine"}
        )

    def add_tables(
        self,
        db_id: str,
        table_names: list[str],
        documents: list[str],
        embeddings: list[list[float]]
    ):
        """
        Запись подготовленных данных в ChromaDB.
        :param db_id: Индекс схемы (напр. 'bike_1').
        :param table_names: Список имен таблиц.
        :param documents: Список строк от Serializer (готовый текст для LLM).
        :param embeddings: Список векторов от Embedder.
        """
        ids = [f"{db_id}.{name}" for name in table_names]
        metadatas = [{"db_id": db_id, "table_name": name} for name in table_names]

        self.table_collection.add(
            ids=ids,
            embeddings=embeddings,
            documents=documents, # Здесь лежат строки для промпта
            metadatas=metadatas
        )
        logger.info(f"Successfully indexed {len(ids)} tables for schema: {db_id}")

    def search_tables(
        self,
        query_embedding: list[float],
        db_id: Optional[str] = None,
        top_k: int = 5
    ) -> list[dict[str, Any]]:
        """
        Поиск релевантных таблиц.
        """
        search_filter = {"db_id": db_id} if db_id else None

        results = self.table_collection.query(
            query_embeddings=[query_embedding],
            n_results=top_k,
            where=search_filter
        )

        # Формируем компактный результат для Cataloger/Agent
        return [
            {"content": doc, "metadata": meta}
            for doc, meta in zip(results["documents"][0], results["metadatas"][0])
        ]
