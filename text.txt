import sqlite3
import pandas as pd
from sqlalchemy import create_engine, text
from src.core.logger import setup_logger

logger = setup_logger("spider_migrator")

class SpiderMigrator:
    def __init__(self, settings, provider):
        self.settings = settings
        self.provider = provider
        # Используем синхронный движок для миграций (DDL операции)
        self.pg_engine = create_engine(self.settings.db_url_sync)

    def migrate_db(self, db_id):
        schema = db_id.lower()
        sqlite_path = self.provider.get_sql_path(db_id)

        # 1. Создание изолированного пространства (Schema)
        with self.pg_engine.connect() as conn:
            conn.execute(text(f"CREATE SCHEMA IF NOT EXISTS {schema}"))
            conn.commit()

        # 2. Перенос данных через SQLite сокет
        with sqlite3.connect(sqlite_path) as sqlite_conn:
            cursor = sqlite_conn.cursor()
            # Получаем список всех таблиц в файле
            cursor.execute("SELECT name FROM sqlite_master WHERE type='table';")
            tables = [t[0] for t in cursor.fetchall() if t[0] != 'sqlite_sequence']

            for table in tables:
                # Читаем из файла в RAM, затем пишем в сетевой сокет Postgres
                df = pd.read_sql_query(f'SELECT * FROM "{table}"', sqlite_conn)
                df.columns = [c.lower() for c in df.columns] # Приводим к нижнему регистру
                df.to_sql(name=table.lower(), con=self.pg_engine, schema=schema, if_exists="replace", index=False)
                logger.info(f"Таблица {schema}.{table.lower()} перенесена")

            # 3. Восстановление метаданных (Ключи и связи)
            self._apply_constraints(schema, tables, cursor)

    def _apply_constraints(self, schema, tables, sqlite_cursor):
        with self.pg_engine.connect() as pg_conn:
            for table in tables:
                t_low = table.lower()

                # Извлекаем Primary Keys из SQLite
                sqlite_cursor.execute(f"PRAGMA table_info('{table}')")
                pks = [row[1].lower() for row in sqlite_cursor.fetchall() if row[5] > 0]
                if pks:
                    pg_conn.execute(text(f'ALTER TABLE {schema}."{t_low}" ADD PRIMARY KEY ({", ".join(pks)})'))

                # Извлекаем Foreign Keys
                sqlite_cursor.execute(f"PRAGMA foreign_key_list('{table}')")
                for fk in sqlite_cursor.fetchall():
                    target_t, from_c, to_c = fk[2].lower(), fk[3].lower(), fk[4].lower()
                    pg_conn.execute(text(f"""
                        ALTER TABLE {schema}."{t_low}"
                        ADD CONSTRAINT fk_{t_low}_{from_c}
                        FOREIGN KEY ({from_c}) REFERENCES {schema}."{target_t}"({to_c})
                    """))
            pg_conn.commit()
            logger.info(f"Связи для схемы {schema} восстановлены")
