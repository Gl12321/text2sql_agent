import asyncio
from src.database.schema_parser import SchemaParser
from src.rag.cataloger import SchemaCataloger
from src.rag.retriver import Retriver
from src.rag.embedder import TableEmbedder
from src.core.logger import setup_logger

logger = setup_logger("ExpertTest")

async def expert_rag_test():
    parser = SchemaParser()
    cataloger = SchemaCataloger()
    retriver = Retriver()
    embedder = TableEmbedder()

    # --- ШАГ 1: Валидация переноса данных ---
    logger.info("Step 1: Validating Data Migration...")

    # Получаем список всех схем и таблиц из Postgres
    all_schemas = await parser.get_all_schemas()
    postgres_stats = {}
    for schema in all_schemas:
        tables = await parser.get_ddl_of_schema(schema)
        postgres_stats[schema] = len(tables)

    # Запускаем индексацию
    await cataloger.index_all_schems()

    # Проверяем наполнение ChromaDB
    for schema, expected_count in postgres_stats.items():
        # Прямой запрос к коллекции для проверки количества записей
        results = retriver.table_collection.get(where={"db_id": schema})
        actual_count = len(results['ids'])

        if actual_count == expected_count:
            logger.info(f"✔ Schema '{schema}': Matches! (Postgres: {expected_count}, Chroma: {actual_count})")
        else:
            logger.error(f"✖ Schema '{schema}': Mismatch! (Postgres: {expected_count}, Chroma: {actual_count})")

    # --- ШАГ 2: Тест семантического поиска (Semantic Accuracy) ---
    logger.info("\nStep 2: Testing Semantic Search Accuracy...")

    # Выбираем одну схему для теста (например, первую)
    if not all_schemas:
        return
    test_db = all_schemas[0]

    # Тестовый запрос, который явно должен вести к одной из таблиц
    # Если в базе есть таблица 'orders', ищем 'sales and transactions'
    test_query = "information about orders and customer transactions"
    query_vector = embedder.get_embeddings([test_query])[0]

    search_results = retriver.search_tables(
        query_embedding=query_vector,
        db_id=test_db,
        top_k=2
    )

    if search_results:
        logger.info(f"✔ Search for '{test_db}' returned {len(search_results)} documents.")
        for i, doc in enumerate(search_results):
            # Твой search_tables возвращает results["documents"][0]
            # Это список строк DDL
            logger.info(f"Top-{i+1} result snippet: {doc[:150]}...")
    else:
        logger.error(f"✖ Search returned NO results for schema '{test_db}'")

if __name__ == "__main__":
    asyncio.run(expert_rag_test())
